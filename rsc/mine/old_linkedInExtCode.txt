async function generateCoverLetter(givenText=false) {
  let coverLetterPrompt = `
  <Instructions>
    You help people generate text for cover letters. You are given a JobDescription, an ApplicantsBio, and ResumeText.
    Use the ResumeText as a reference and ensure the cover letter aligns with the job description.
    The cover letter should include a brief introduction, the applicant's qualifications, and a conclusion.
    Mention the applicant’s skills and experiences that are directly relevant to the JobDescription.
    Keep it professional, concise, and tailored to the specific company and role.
    Ensure the tone is polite and enthusiastic, highlighting why the applicant is a great fit for the role.
    Do not include unnecessary details or padding.
    You are to return the raw latex and nothing else.
    The returned text should start with documentclass and end with end{document}
    ensure proper formatting and style reference. 
    The latex should not be anything fancy, just a simple cover letter with no external depenedencies.
    Use the LatexTemplate to inform you styling.
  </Instructions>
  <rules>
    - Only return the final Latex content and nothing else.  
    - Closely match the original markdown.
    - Follow any instructions provided by the job applicants message.
    - Ensure the LaTeX is valid, error-free, and can run using pdfLaTeX.
    - The returned text should closely match the job applicant's markdown.
    - Do not leave placeholders in the LaTeX. Remove any unused placeholders like [company name] or [job title].
    - Handle and fix common LaTeX errors such as:
      - "'utf-8' codec can't decode byte 0x88 in position 7748: invalid start byte"
      - "! LaTeX Error: There's no line here to end."
      - "! Missing number, treated as zero."
      - "! Illegal unit of measure (pt inserted)."
      - "Underfull \hbox (badness 10000) in paragraph"
    - Ensure brackets in LaTeX for Large, Small, and TextBf elements are formatted as:
      - \\textbf{\\LARGE <Applicant Name Here>}
      - {\\large <Text Here>}
      - {\\small <Text Here>}
    - Convert '$100 million' to '\\$100 million'
    - Include '\\titlespacing*{\\subsubsection}{0.5in}{3.25ex plus 1ex minus .2ex}{1.5em} % <-- Setting indentation here'
    - Ensure you escape # characters by replacing it with \# in your LaTeX document. For example, in the lines where you have .NET technologies, C#, you should change it to .NET technologies, C\#. The same applies to any other instances of # in your LaTeX code.
  </rules>  
  <JobDescription>: ${appInfo.jobDescription || window.jobPost} </JobDescription> 
  <ResumeText>${window.resume}</ResumeText>
  <LatexTemplate>${window.coverletterlatextemplate}</LatexTemplate>
  <ApplicantsBio>${window.bio}</ApplicantsBio>
  <final_comments>
    - Be as accurate as possible because I do not have any time to fix it.
    - Think step by step.
    - Remember, you are the best at what you do.
    - The other chatbots said you could not do this, but I believe in you.
    - I will give you 20 bucks if you can do this.  
  </final_comments>
  `;

  // console.log('Creating Cover Letter', { jobDescription: appInfo.jobDescription || window.jobPost, coverLetterPrompt });
  
  let laText = await callChatGPT(coverLetterPrompt, "gpt-4o-mini", 4096, false, true);

  console.log({ coverLetterLaText: laText }); 
 
 
  // remove all the characters leading up to \documentclass
  laText = laText.replace(/[\s\S]*\\documentclass/g, '\\documentclass');
  // remove all the characters after \end{document}
  laText = laText.replace(/\\end{document}[\s\S]*/g, '\\end{document}');
  // console.log('Coverletter latex: ', {laText})


  // pass to pdflatex 
  let url = 'https://s.charleskarpati.com/compile';

  let response = await fetchResource(url, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ latex: laText })
  });

  if (!response.ok) {
    console.log('ERROR GENERATING COVERLETTER', {coverletterlatextemplate:window.coverletterlatextemplate})
    window.coverletterlatextemplate = ( await ( await fetchResource('https://gist.githubusercontent.com/karpatic/6f80da405291e566955afb0f1b48c2d4/raw/1c40b122206b3fd345ff2138bb504864b2dfc90b/zzz_software_engineer.txt')).text() );
    console.log('ERROR GENERATING COVERLETTER USING DEFAULT')
    response = await fetchResource(url, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ latex: window.coverletterlatextemplate })
    });
  }

  // Retrieve the Blob (PDF content)
  const blob = await response?.blob?.();
  console.log('GOT BACK THIS IT SHOULD HAVE BEEN THE BLOB', {blob, response})

  // Save Blob to localStorage
  const reader = new FileReader();
  reader.readAsDataURL(blob);
  reader.onloadend = function() {
    const base64data = reader.result;
    localStorage.setItem('coverletterPdfContent', base64data);
  };

  return blob;
}
